### Title
Revoked contract pusher can still push arbitrary prices into creator's namespace — (`File: smart-contracts-poc/contracts/oracles/compressed/CompressedOracle.sol`)

---

### Summary

`allowContractPushers` gates registration on a live `isPusher(creator)` staticcall, but the `fallback()` push path never re-checks `isPusher`. Once a contract pusher is registered, it can push any price into the creator's namespace indefinitely — even after the pusher contract has revoked its own consent — because the only runtime check is `namespaceRemapping[msg.sender] != address(0)`.

---

### Finding Description

`allowContractPushers` is documented as using a "LIVE `isPusher(creator)` staticcall" as the consent proof, explicitly stating "there is nothing to replay and no deadline is needed." This implies `isPusher` is the ongoing authorization gate. In practice it is only a one-time registration gate. [1](#0-0) 

After registration, `namespaceRemapping[pusher] = creator` is written and never automatically cleared. The `fallback()` push path reads only that mapping: [2](#0-1) 

There is no call to `isPusher` inside `fallback()`. The only guards are:

1. `timestampMs.revertIfAfterBlockTimeWithDrift(MAX_TIME_DRIFT)` — future-drift bound
2. `timestampMs.isAfter(oldTimestampMs)` — monotonicity [3](#0-2) 

Neither guard checks whether the pusher contract still consents. A pusher whose `isPusher(creator)` now returns `false` (revoked at the pusher-contract level) can still write any price into the creator's slot, subject only to the timestamp being strictly greater than the last stored value.

This is the direct analog of the Alchemix killed-gauge bug: `killGauge` sets `isAlive = false` but `_updateFor` never reads `isAlive`, so the killed gauge keeps accruing. Here, the pusher contract signals revocation via `isPusher → false`, but `fallback()` never reads `isPusher`, so the revoked pusher keeps writing prices.

---

### Impact Explanation

Bad prices written by a revoked (or compromised) contract pusher flow directly to pool swaps through `PriceProvider._getBidAndAskPrice()`: [4](#0-3) 

`PriceProvider` calls `IPricedOracle(address(offchainOracle)).price(offchainFeedId, msg.sender)`, which for `CompressedOracle` is a permissionless view that reads directly from the slot the attacker just wrote. The staleness check (`_isStale`) and price-guard check are the only downstream filters; a freshly pushed, in-range but manipulated price passes both. Traders executing swaps against the pool receive execution at the attacker-controlled bid/ask, causing direct loss of principal.

---

### Likelihood Explanation

The trigger is a semi-trusted contract pusher whose `isPusher` consent has been revoked — either because the pusher contract's admin deliberately withdrew consent, or because the pusher contract was upgraded/compromised. The creator must have previously called `allowContractPushers` for the pusher. The creator can mitigate by calling `removePushers`, but there is a window between revocation and removal during which the pusher can write arbitrary prices. No privileged oracle-admin action is required from the attacker side; the registered pusher address is sufficient.

---

### Recommendation

Re-check `isPusher(creator)` inside `fallback()` for every push from a contract pusher, mirroring the live-consent model the NatSpec describes:

```solidity
fallback() override external {
    address creator = namespaceRemapping[msg.sender];
    if (creator != address(0)) {
        // Re-validate live consent on every push
        (bool ok, bytes memory res) =
            msg.sender.staticcall(abi.encodeWithSignature("isPusher(address)", creator));
        require(ok && abi.decode(res, (bool)), "pusher consent revoked");
    } else {
        creator = msg.sender;
    }
    // ... rest of push logic
}
```

Alternatively, emit an event and revert (rather than silently clearing `namespaceRemapping`) when `isPusher` returns false, so the creator is notified on-chain.

// Attacker-controlled pusher contract
contract MaliciousPusher {
    address public creator;
    bool public consent;

    constructor(address _creator) {
        creator = _creator;
        consent = true; // returns true at registration time
    }

    // Called by CompressedOracle.allowContractPushers
    function isPusher(address _creator) external view returns (bool) {
        return consent && _creator == creator;
    }

    // Attacker revokes consent at pusher-contract level
    function revokeConsent() external { consent = false; }

    // Push bad price into creator's namespace — still works after revokeConsent()
    function pushBadPrice(address oracle, uint256 slotWord) external {
        (bool ok,) = oracle.call(abi.encodePacked(slotWord));
        require(ok);
    }
}

// Attack sequence:
// 1. creator calls oracle.allowContractPushers([address(maliciousPusher)])
//    → isPusher(creator) returns true → namespaceRemapping[pusher] = creator
// 2. maliciousPusher.revokeConsent()
//    → isPusher(creator) now returns false
//    → but namespaceRemapping[pusher] is still set to creator
// 3. maliciousPusher.pushBadPrice(oracle, craftedSlotWord)
//    → fallback() reads namespaceRemapping[pusher] == creator (not zero)
//    → no isPusher re-check → bad price written to creator's slot
// 4. PriceProvider.getBidAndAskPrice() reads the manipulated price
//    → pool swap executes at attacker-controlled bid/ask
//    → trader loses principal

    /// @notice Contract-pusher variant: consent is proven by a LIVE `isPusher(creator)`
    ///         staticcall instead of a signature, so there is nothing to replay and no
    ///         deadline is needed.
    function allowContractPushers(address[] calldata pushers) external {
        uint256 l = pushers.length;
        for (uint256 i; i < l; i++) {
            address pusher = pushers[i];

            if (pusher == msg.sender) {
                revert NoSelfRemapping();
            }

            (bool ok, bytes memory res) = pusher.staticcall(abi.encodeWithSignature("isPusher(address)", msg.sender));
            require(ok);
            bool allowed = abi.decode(res, (bool));
            require(allowed);

            namespaceRemapping[pusher] = msg.sender;
            emit PusherAuthorized(pusher, msg.sender);
        }

    fallback() override external {
        uint256 end;
        uint256 namespace;

        address creator = namespaceRemapping[msg.sender];
        if (creator == address(0)) creator = msg.sender;

        assembly ("memory-safe") {
            end := calldatasize()
            namespace := shl(96, creator) // [creator:20][zeros:12]
        }

        // 4 * 6 + 7 + 1 = 32 bytes per slot
        if (end == 0 || end % 32 != 0) revert BadCalldataLength();

        for (uint256 ptr = 0; ptr < end; ptr += 32) {
            uint256 word;
            assembly ("memory-safe") {
                word := calldataload(ptr)
            }
            // casting to 'uint8' is safe we want LSB
            // forge-lint: disable-next-line(unsafe-typecast)
            uint8 slotId = uint8(word);
            TimeMs timestampMs = toTimeMs(word >> 8 & X56);
            timestampMs.revertIfAfterBlockTimeWithDrift(MAX_TIME_DRIFT);
            bytes32 key = bytes32(namespace | uint256(slotId));
            uint256 old = uint256(_loadStorage(key));
            TimeMs oldTimestampMs = toTimeMs(old >> 8 & X56);

            bool newer = timestampMs.isAfter(oldTimestampMs);
            if (!newer) continue;

            _writeStorage(key, bytes32(bytes32(word & ~uint256(0xff))));
